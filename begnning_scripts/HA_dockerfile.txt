ARG BUILD_FROM=python:3.9-slim
FROM $BUILD_FROM

# Install system dependencies (jq added for parsing HA config)
RUN apt-get update && apt-get install -y \
    libgl1-mesa-glx \
    libglib2.0-0 \
    jq \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Install Python requirements
RUN pip install --no-cache-dir torch torchvision --index-url https://download.pytorch.org/whl/cpu
RUN pip install --no-cache-dir transformers opencv-python paho-mqtt matplotlib numpy

# Copy pipeline files
COPY segformer_final_script.py /app/
COPY run.sh /app/

# Make the wrapper script executable
RUN chmod a+x /app/run.sh

# Hand over execution to the wrapper
CMD [ "/app/run.sh" ]
